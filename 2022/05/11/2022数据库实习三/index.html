static bool InsertData(SQLHDBC hdbc, int mFID, float mX, float mY, int mcolor, int mpattern) {

    SQLHSTMT hstmt = SQL_NULL_HSTMT;
    // Allocate execute statement handle.
    SQLRETURN ret = SQLAllocHandle(SQL_HANDLE_STMT, hdbc, &hstmt);
    if (!SQL_SUCCEEDED(ret)) {
        std::cout << "Failed to allocate execute statement handle!" << std::endl;
        return false;
    }
    SQLSMALLINT FID = 0;
    SQLFLOAT X = 0;
    SQLFLOAT Y = 0;
    SQLSMALLINT color = 0;
    SQLSMALLINT pattern = 0;

    SQLWCHAR insert_sql[256] = L"insert into PNT_Table(FID,X, Y,color, pattern) values(?,?,?,?,?)";
    ret = SQLPrepare(hstmt, insert_sql, SQL_NTS);
    if (ret != SQL_SUCCESS) {
        std::cout << "Failed to prepare sql!" << std::endl;
    }
    ret = SQLBindParameter(hstmt, 1, SQL_PARAM_INPUT, SQL_C_SHORT, SQL_SMALLINT, 0, 0, (void*)&FID, 0, 0);
    ret = SQLBindParameter(hstmt, 2, SQL_PARAM_INPUT, SQL_C_DOUBLE, SQL_FLOAT, 0, 0, (void*)&X, 0, 0);
    ret = SQLBindParameter(hstmt, 3, SQL_PARAM_INPUT, SQL_C_DOUBLE, SQL_FLOAT, 0, 0, (void*)&Y, 0, 0);
    ret = SQLBindParameter(hstmt, 4, SQL_PARAM_INPUT, SQL_C_SHORT, SQL_SMALLINT, 0, 0, (void*)&color, 0, 0);
    ret = SQLBindParameter(hstmt, 5, SQL_PARAM_INPUT, SQL_C_SHORT, SQL_SMALLINT, 0, 0, (void*)&pattern, 0, 0);

    FID = mFID;
    X = mX;
    Y = mY;
    color = mcolor;
    pattern = mpattern;

    ret = SQLExecute(hstmt);
    if (ret != SQL_SUCCESS) {
        std::cout << "Failed to execute sql!" << std::endl;
    }
    else
    {
        std::cout << "Inserted successfully!" << std::endl;
    }
    ret = SQLFreeHandle(SQL_HANDLE_STMT, hstmt);
    if (!SQL_SUCCEEDED(ret)) {
        std::cout << "Failed to release execute statement handle!" << std::endl;
    }
    std::cout << "insert information: FID:" << mFID << ",横坐标:" << mX << ",纵坐标:" << mY << ",颜色号:" << mcolor << ", 图案编号:" << mpattern << std::endl;

    return true;
}

static bool UpdateDataSQL(SQLHDBC hdbc, float oldX ,float oldY, float mX, float mY) {

    // Allocate execute statement handle.

        return false;
    }
    //t旧

    SQLFLOAT tY = 0;
    SQLFLOAT X = 0;
    SQLFLOAT Y = 0;
    /*SQLWCHAR insert_sql[256] = L"UPDATE PNT_Table SET X=9 ,Y =6 ,color=8 ,pattern=99 where FID = 3";*/
    SQLWCHAR insert_sql[256] = L"UPDATE PNT_Table SET X= ? ,Y = ?  where X= ? AND Y= ?";
    ret = SQLPrepare(hstmt, insert_sql, SQL_NTS);
    if (ret != SQL_SUCCESS) {
        std::cout << "Failed to prepare sql!" << std::endl;
    }
    ret = SQLBindParameter(hstmt, 3, SQL_PARAM_INPUT, SQL_C_DOUBLE, SQL_FLOAT, 0, 0, (void*)&tX, 0, 0);
    ret = SQLBindParameter(hstmt, 4, SQL_PARAM_INPUT, SQL_C_DOUBLE, SQL_FLOAT, 0, 0, (void*)&tY, 0, 0);
    ret = SQLBindParameter(hstmt, 1, SQL_PARAM_INPUT, SQL_C_DOUBLE, SQL_FLOAT, 0, 0, (void*)&X, 0, 0);
    ret = SQLBindParameter(hstmt, 2, SQL_PARAM_INPUT, SQL_C_DOUBLE, SQL_FLOAT, 0, 0, (void*)&Y, 0, 0);
    tX = oldX;
    tY = oldY;
    X = mX;
    Y = mY;
    ret = SQLExecute(hstmt);
    if (ret != SQL_SUCCESS) {
        std::cout << "Failed to execute sql!" << std::endl;
    }
    else
    {
        std::cout << "Updated successfully!" << std::endl;
    }
    // Release execute statement handle.
    ret = SQLFreeHandle(SQL_HANDLE_STMT, hstmt);
    if (!SQL_SUCCEEDED(ret)) {
        std::cout << "Failed to release execute statement handle!" << std::endl;
    }

    return true;
}

4.得到点数据：原工程文件”MapEditorView.cpp”添加FindData函数，在修改点参数时调用查找鼠标点击位置最近的点ID

static int FindData(SQLHDBC hdbc,float mX,float mY) {
    SQLHSTMT hstmt = SQL_NULL_HSTMT;

    // Allocate execute statement handle.
    SQLRETURN ret = SQLAllocHandle(SQL_HANDLE_STMT, hdbc, &hstmt);
    if (!SQL_SUCCEEDED(ret)) {
        std::cout << "Failed to allocate execute statement handle!" << std::endl;
        return false;
    }

    SQLSMALLINT FID = 0;
    SQLFLOAT X = 0;
    SQLFLOAT Y = 0;
    
    SQLWCHAR select_sql[256] = L"select FID from PNT_Table WHERE X=? AND Y=?";
    ret = SQLPrepare(hstmt, select_sql, SQL_NTS);
    if (!SQL_SUCCEEDED(ret)) {
        std::cout << "Failed to query data!" << std::endl;
    }
    /*SQLBindCol(hstmt, 1, SQL_C_SHORT, (void*)&FID, sizeof(FID), NULL);
    SQLBindCol(hstmt, 2, SQL_C_FLOAT, (void*)&X, sizeof(X), NULL);
    SQLBindCol(hstmt, 3, SQL_C_FLOAT, (void*)&Y, sizeof(Y), NULL);*/
    //ret = SQLBindParameter(hstmt, 1, SQL_PARAM_INPUT, SQL_C_SHORT, SQL_SMALLINT, 0, 0, (void*)&FID, 0, 0);
    ret = SQLBindParameter(hstmt, 1, SQL_PARAM_INPUT, SQL_C_DOUBLE, SQL_FLOAT, 0, 0, (void*)&X, 0, 0);
    ret = SQLBindParameter(hstmt, 2, SQL_PARAM_INPUT, SQL_C_DOUBLE, SQL_FLOAT, 0, 0, (void*)&Y, 0, 0);
    X = mX;
    Y = mY;
    // Release execute statement handle.
    SQLBindCol(hstmt, 1, SQL_C_SHORT, (void*)&FID, sizeof(FID), NULL);
    ret = SQLExecute(hstmt);
    while (SQLFetch(hstmt) != SQL_NO_DATA) {
        std::cout << "FID: " << FID;
        return FID;
    }
    return true;
}
