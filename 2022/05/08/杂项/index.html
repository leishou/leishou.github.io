#ifndef BLOCK_H
#define BLOCK_H
#include<iostream>
#include<process.h>
#include <memory.h>
#include<memory>
#include<stdio.h>
using namespace std;
struct BlockNode {
    char val;  // 程序名
    BlockNode* next;  // 指向下一个节点的指针
    BlockNode();
    BlockNode(int x) : val(x), next(NULL) {}  // 节点的构造函数
    int size;//内存块大小
    int pos;//起始位置
};
class BlockList {
public:
    BlockList() {
        first = NULL;
    }
    void init( int tmp);//初始化
    void assign(BlockNode* q, char val, int size);
    int ff(int tmp,char val,int size);
    int bf(int tmp, char val, int size);
    int wf(int tmp,char val,int size);
    void f_assign(int tmp);
    void Main( int tmp);
    void showState();//查看内存状态
    int  releaseSource(int tmp,char val,int size);//撤销进程，释放资源

private:
    BlockNode* first;
    BlockNode* p;
};
BlockNode* base;
int main()
{
    char choice;
    int tmp;
    BlockList cs;
    while (1)
    {
        system("cls");
        cout << "演示" <<endl ;
        cout << "1.静态分配 2.动态分配 3 退出程序" << endl;

        cin >> choice;
        if (choice == '1')tmp = 0;
        else if (choice == '2')tmp = 1;
        else if (choice == '3')break;
        cs.Main(tmp);
    }
    return 0;
}

void BlockList::init( int tmp)
{
    BlockNode* p, * q;
    if (base != NULL)
    {
        p = base;
        while (p)
        {
            q = p->next;
            delete p;
            p = q;
        }
    }
    base = new BlockNode;//操作系统 大小24k
    base->pos = 0;
    base->val = 'o';//代表操作系统
    base->size = 4;//4k
    if (tmp == 0)//静态处理
    {
        //1
        p = base;
        q = new BlockNode;
        q->pos = 4;
        q->val = 0;
        q->size = 20;
        p->next = q;
        p = q;
        //2
        q = new BlockNode;
        q->pos = 24;
        q->val = 0;
        q->size = 40;
        p->next = q;
        p = q;
        //3
        q = new BlockNode;
        q->pos = 64;
        q->val = 0;
        q->size = 50;
        p->next = q;
        p = q;
        //4
        q = new BlockNode;
        q->pos = 124;
        q->val = 0;
        q->size = 30;
        p->next = q;
        p = q;

        //5

    }
    else

    }
}

void BlockList::assign(BlockNode* q, char val, int size)
{

    //q 待检测的块
    BlockNode* p = q->next;
    BlockNode* temp = new BlockNode;
    temp = new BlockNode;
    temp->pos = p->pos;
    temp->size = p->size;
    q->next = temp;
    if (p->size < size)
    {
        p->size = p->size - size;
        p->pos = temp->pos + temp->size;
        temp->next = p;
    }
    else if(p->size=size)
    {
        temp->next = p->next;
        delete p;
    }
    else
    {
        cout << "无法分配！该块空间不足！";
    }
}

int BlockList::ff(int tmp, char val, int size)
{
    BlockNode* p = base;
    BlockNode* q = p;
    while (p)
    {
        if (p->size < size || p->val != 0)
        {
            q = p;
            p = p->next;
        }
        else break;
    }
    if (p == NULL)return 0;//未找到
    else
    {
        if (tmp == 0)p->val = val;//进程进驻内存

        else assign(q, val, size);
    }
    return 1;//返回1
}

int BlockList::bf(int tmp, char val, int size)
{
    BlockNode* p = base, * temp = NULL, * q = NULL, * front = NULL;
    int min = 256;
    while (p)
    {
        if (p->val == 0 && p->size >= size && min>p->size)
        {
            min = p->size;//更新最小值
            front = q;
            temp = p;
        }
        q = p;
        p = p->next;
    }
    if (temp == NULL)return 0;//
    else
    {
        if (tmp == 0)temp->val = val;//静态进驻

        else assign(front, val, size);
    }
    return 1;
}

int BlockList::wf(int tmp, char val, int size)
{
    BlockNode* p = base, * temp = NULL, * q = NULL, * front = NULL;
    int max = 0;
    while (p)
    {
        if (p->val == 0 && p->size >= size && max < p->size)
        {
            max = p->size;//更新最小值
            front = q;
            temp = p;
        }
        q = p;
        p = p->next;
    }
    if (temp == NULL)return 0;//
    else
    {
        if (tmp == 0)temp->val = val;//静态进驻

        else assign(front, val, size);
    }
    return 1;

}

void BlockList::f_assign(int tmp)
{
    int size, result = -1;
    char chose, val;
    cout << "请输入进程名:";
    cin >> val;
    cout << "请输入进程分配内存的大小：";
    cin >> size;
    cout << "请选择分配算法：1.首次 2 最佳 3 最差" << endl;
    cin >> chose;
    switch (chose)
    {
    case '1':
        result = ff(chose, val, size);
        break;
    case '2':
        result = bf(chose, val, size);
        break;
    case '3':
        result = wf(chose, val, size);
        break;
    }
    if (result == 0)cout << "进程" << val << "分配失败！" << endl;
    else if (result == 1)cout << "进程" << val << "分配成功！" << endl;
    else cout << "非法输入！" << endl;

}

void BlockList::Main(int tmp)
{
    char choice;
    init(tmp);
    while (1)
    {
        system("cls");

        showState();
        cout << "请选择操作： \b 1.建立进程并分配" << endl;
        cin >> choice;
        if (choice == '1')
        {
            f_assign(tmp);
            system("pause");
        }
    }
}

int BlockList::releaseSource(int tmp, char val, int size)
{
    BlockNode* p = base, * q = NULL;
    while (p)
    {
        if (p->val != val)
        {
            q = p;
            p = p->next;
        }
        else break;
    }
    if (p == NULL) return 0;//未找到
    //q  p
    else
    {
        if (tmp == 0)p->val = 0;
        else
        {
            if (q->val!= 0)
            {    //1 前非 后非
                if (p->next != NULL || p->next->val != 0)
                {
                    p->val = 0;
                }
                //2 前非 后空
                else
                {
                    q->next = p->next;
                    p->next->pos = p->pos;
                    p->next->size = p->size +p->next->size;

                    delete p;
                }
            }
            else
            {   //3 前空 后非
                if (p->next == NULL || p->next->val != 0)
                {
                    q->next = p->next;
                    q->size = q->size + p->size;
                    delete p;
                }
                //4 前空 后空
                else
                {
                    q->next = p->next->next;
                    q->size = q->size + p->size+p->next->size;

                    delete p;
                    delete p->next;
                }
            }
        }
    }
    return 1;
}
