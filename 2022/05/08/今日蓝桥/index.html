#include<iostream>
#include<bits/stdc++.h>
using namespace std;
const int N=10010;
char s[N];
int dp[N];
int ptr[30];
int main()
{
    cin>>s+1;
    long long ans=0;
    int len=strlen(s+1);
    for(int i=1;i<=len;i++)
    {
        int t=s[i]-'a';
        dp[i]=dp[i-1]+i-ptr[t];
        ptr[t]=i;
        ans+=dp[i];
    }
    cout<<ans;
    return 0;
}

    vec.push_back({i,j});
    
    map<pair<double,double>,bool>ma;
    for(int i=0;i<vec.size();i++)
    for(int j=i+1;j<vec.size();j++)
    {
        
        int x1=vec[i].first,x2=vec[j].first;
        int y1=vec[i].second,y2=vec[j].second;
        if(x1!=x2&&y1!=y2){
            double k,b;
            k= double(y2-y1)/(x2-x1);
            b=double(x1*y2-x2*y1)/(x1-x2);
            ma[{k,b}]=true;
        }
    }
    cout<<ma.size()+X+Y<<endl;
    return 0;
 }

路径 dijkstra()

#include<bits/stdc++.h>
#include<iostream>
using namespace std;
int gcd(int a,int b)
{
    return b?gcd(b,a%b):a;
}
int stimes(int a,int b)
{
    return a*b/gcd(a,b);
}
const int N=2022,M=22;
int g[N][N],dist[N];
bool st[N];
void dijkstra()
{
    dist[1]=0;
    for(int i=1;i<N;i++)
    {
        int t=-1;
    for(int j=1;j<N;j++)
    if(!st[j]&&(t==-1||dist[j]<dist[t]))
    t=j;
    st[t]=true;
    
    for(int j=1;j<N;j++)
    dist[j]=min(dist[j],dist[t]+g[t][j]);
}

}
int main()
{
    memset(dist,0x3f,sizeof dist);
    memset(g,0x3f,sizeof g);
    
    for(int i=1;i<N;i++)
    for(int j=i+1;j<N&&j-i<M;j++)
    g[i][j]=g[j][i]=stimes(i,j);
    
    dijkstra();
    
    cout<<dist[2021]<<endl;
    return 0;
}

#include<bits/stdc++.h>
#include<iostream>
using namespace std;
const int N=22,M=1<<N;
int gcd(int a,int b)
{
    return b?gcd(b,a%b):a;
}
bool g[N][N];
long long f[M][N];

int main()
{
    for(int i=1;i<N;i++)
    for(int j=1;j<N;j++)
    if(gcd(i,j)==1)
    g[i][j]=true;
    
    f[2][1]=1;
    for(int i=1;i<=M-2;i++)//遍历 
    for(int j=1;j<N;j++)//到达的点 
    if(i>>j&1)
    for(int k=1;k<N;k++)
    if(g[k][j]&&i-(1<<j)>>k&1) 
    f[i][j]+=f[i-(1<<j)][k];
    
    long long ans=0;
    for(int i=1;i<N;i++)
    ans+=f[M-2][i];
    
    cout<<ans<<endl;
    
    return 0;
}

#include<bits/stdc++.h>
#include<iostream>
using namespace std;
const int maxn=10052;
int fa[maxn],n;
vector<int>u[maxn];

int dfs(int x){
    int ret=1;
    
    for(int i=0;i<u[x].size();i++)
    {
        int temp =1+dfs(u[x][i])+u[x].size()-1;
        ret =max(temp,ret);
    }
    return ret;
}
int main()
{
    cin>>n;
    for(int i=0;i<n-1;i++)
    {
        cin>>fa[i];
        u[fa[i]].push_back(i+2);
    }
    cout<<dfs(1)-1<<endl;
    return 0;
}

#include<iostream>
#include<queue>

using namespace std;
bool visit[2021] = { true };

int main()
{
    queue<int>cs;
    cs.push(2021);
    int time = 0;
    while (!cs.empty())
    {
        int size = cs.size();
        while (size-- > 0)
        {
            int cur = cs.front();
            cs.pop();
            if (cur == 1)
            {
                cout << time;
                return 0;
            }
            if (!visit[cur + 1])
            {
                visit[cur + 1] = true;
                cs.push(cur + 1);
            }
            if (!visit[cur - 1])
            {
                visit[cur - 1] = true;
                cs.push(cur - 1);
            }
            if (cur % 2 == 0 && !visit[cur / 2])
            {
                visit[cur / 2] = true;
                cs.push(cur / 2);
            }

        }
        time++;
    }
    return 0;
}

#include<iostream>

typedef  long long LL;
using namespace std;
const int N=1000000+10;
int arr[N];
void cal(LL n)
{
    for(LL i=1;i<=sqrt(n);i++){
        if(n%i==0){
            arr[i]++;
            if(n!=i*i){
                arr[n/i]++;
            }
        }
    }
}
