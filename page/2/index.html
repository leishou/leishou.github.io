struct BlockNode {
    char val;  // 程序名
    BlockNode* next;  // 指向下一个节点的指针
    BlockNode();
    BlockNode(int x) : val(x), next(NULL) {}  // 节点的构造函数
    int size;//内存块大小
    int pos;//起始位置
};
class BlockList {
public:
    BlockList() {
        first = NULL;
    }

    void assign(BlockNode* q, char val, int size);

    int  releaseSource(int tmp,char val,int size);//撤销进程，释放资源

private:
    BlockNode* first;
    BlockNode* p;
};
BlockNode* base;

void BlockList::assign(BlockNode* q, char val, int size)
{

    //q 待检测的块
    BlockNode* p = q->next;
    BlockNode* temp = new BlockNode;
    temp = new BlockNode;
    temp->pos = p->pos;
    temp->size = p->size;
    q->next = temp;
    if (p->size < size)
    {
        p->size = p->size - size;
        p->pos = temp->pos + temp->size;
        temp->next = p;
    }
    else if(p->size=size)
    {
        temp->next = p->next;
        delete p;
    }
    else
    {
        cout << "无法分配！该块空间不足！";
    }
}

int BlockList::ff(int tmp, char val, int size)
{
    BlockNode* p = base;
    BlockNode* q = p;
    while (p)
    {
        if (p->size < size || p->val != 0)
        {
            q = p;
            p = p->next;
        }
        else break;
    }
    if (p == NULL)return 0;//未找到
    else
    {
        if (tmp == 0)p->val = val;//进程进驻内存

        else assign(q, val, size);
    }
    return 1;//返回1
}

int BlockList::bf(int tmp, char val, int size)
{
    BlockNode* p = base, * temp = NULL, * q = NULL, * front = NULL;
    int min = 256;
    while (p)
    {
        if (p->val == 0 && p->size >= size && min>p->size)
        {
            min = p->size;//更新最小值
            front = q;
            temp = p;
        }
        q = p;
        p = p->next;
    }
    if (temp == NULL)return 0;//
    else
    {
        if (tmp == 0)temp->val = val;//静态进驻

        else assign(front, val, size);
    }
    return 1;
}

int BlockList::wf(int tmp, char val, int size)
{
    BlockNode* p = base, * temp = NULL, * q = NULL, * front = NULL;
    int max = 0;
    while (p)
    {
        if (p->val == 0 && p->size >= size && max < p->size)
        {
            max = p->size;//更新最小值
            front = q;
            temp = p;
        }
        q = p;
        p = p->next;
    }
    if (temp == NULL)return 0;//
    else
    {
        if (tmp == 0)temp->val = val;//静态进驻

        else assign(front, val, size);
    }
    return 1;

}

int BlockList::releaseSource(int tmp, char val, int size)
{
    BlockNode* p = base, * q = NULL;
    while (p)
    {
        if (p->val != val)
        {
            q = p;
            p = p->next;
        }
        else break;
    }
    if (p == NULL) return 0;//未找到
    //q  p
    else
    {
        if (tmp == 0)p->val = 0;
        else
        {
            if (q->val!= 0)
            {    //1 前非 后非
                if (p->next != NULL || p->next->val != 0)
                {
                    p->val = 0;
                }
                //2 前非 后空
                else
                {
                    q->next = p->next;
                    p->next->pos = p->pos;
                    p->next->size = p->size +p->next->size;

                    delete p;
                }
            }
            else
            {   //3 前空 后非
                if (p->next == NULL || p->next->val != 0)
                {
                    q->next = p->next;
                    q->size = q->size + p->size;
                    delete p;
                }
                //4 前空 后空
                else
                {
                    q->next = p->next->next;
                    q->size = q->size + p->size+p->next->size;

                    delete p;
                    delete p->next;
                }
            }
        }
    }
    return 1;
}

操作系统以扇区为单位对磁盘进行读/写操作

在操作系统中,FCB是指 文件控制块

磁带、磁盘这样的存储设备都是以块为单位与内存进行信息交换的。

在UNIX系统中，磁盘空闲空间管理所采用的方法是（）

空闲块成组链接法

逻辑文件是从用户观点看的文件组织形式

位示图可用于磁盘空间的管理

设文件索引结点中有8个地址项，每个地址项大小为4字节，其中5个地址项为直接地址索引，2个地址项是一级间接地址索引，1个地址项是二级间接地址索引，磁盘索引块和磁盘数据块大小均为1KB。则可表示的单个文件最大长度是多少KB？

解：磁盘索引块为1KB字节，每个地址项大小为4字节，故每个磁盘索引块可存放1024/4=256个物理地址块。又因为文件索引节点中有8个地址项，其中5个地址项为直接地址索引，这意味着逻辑块号为0—4的为直接地址索引；2个地址项是一级间接地址索引，这意味着第一个地址项指出的物理块中存放逻辑块号为5—260的物理块号，第二个地址项指出的物理块中存放逻辑块号为261—516的物理块号；1个地址项是二级间接地址索引，该地址项指出的物理块存放了256个间接索引表的地址，这256个间接索引表存放逻辑块号为517—66052的物理块号（256*256=65536个）。单个文件的逻辑块号范围是0—66052，而磁盘数据块大小为1KB，所以单个文件最大长度为：66053KB。
